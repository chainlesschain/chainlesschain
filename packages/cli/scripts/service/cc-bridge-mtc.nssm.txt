NSSM (Non-Sucking Service Manager) commands for registering
`cc crosschain mtc-serve` as a Windows service.

Download NSSM: https://nssm.cc/download

Run all commands in an elevated (Administrator) PowerShell or cmd.

# Install service
nssm install cc-bridge-mtc "C:\Program Files\nodejs\node.exe"
nssm set cc-bridge-mtc AppParameters "C:\path\to\cc\bin\chainlesschain.js crosschain mtc-serve --interval 60 --json"
nssm set cc-bridge-mtc AppDirectory "%USERPROFILE%"
nssm set cc-bridge-mtc DisplayName "ChainlessChain — Cross-chain bridge MTC"
nssm set cc-bridge-mtc Description "Periodic batch closer for cross-chain MTC envelopes"
nssm set cc-bridge-mtc Start SERVICE_AUTO_START

# Logging — separate stdout / stderr files (rotate manually or via Task Scheduler)
nssm set cc-bridge-mtc AppStdout "%USERPROFILE%\.chainlesschain\logs\cc-bridge-mtc.out.log"
nssm set cc-bridge-mtc AppStderr "%USERPROFILE%\.chainlesschain\logs\cc-bridge-mtc.err.log"
nssm set cc-bridge-mtc AppRotateFiles 1
nssm set cc-bridge-mtc AppRotateOnline 1
nssm set cc-bridge-mtc AppRotateBytes 10485760

# Restart on failure (3 retries, 5s delay)
nssm set cc-bridge-mtc AppExit Default Restart
nssm set cc-bridge-mtc AppRestartDelay 5000

# Graceful stop — NSSM sends Ctrl-C (SIGINT-equivalent) by default;
# the cc daemon clears its interval timer and exits 0.
nssm set cc-bridge-mtc AppStopMethodSkip 0
nssm set cc-bridge-mtc AppStopMethodConsole 10000

# Start it
nssm start cc-bridge-mtc

# Stop / remove later
# nssm stop cc-bridge-mtc
# nssm remove cc-bridge-mtc confirm
