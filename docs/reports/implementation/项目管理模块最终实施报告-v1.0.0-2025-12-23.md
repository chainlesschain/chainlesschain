```javascript
// 首次加载项目时，索引所有文件
await window.electronAPI.project.indexFiles(projectId, {
  forceReindex: false
});

// AI对话时，使用RAG增强
async function chatWithRAG(userMessage) {
  // 1. RAG检索
  const context = await window.electronAPI.project.ragQuery(
    projectId,
    userMessage,
    {
      projectLimit: 5,
      knowledgeLimit: 3,
      useReranker: true
    }
  );

  // 2. 构建增强提示词
  const enhancedPrompt = `
项目上下文:
${context.summary}

用户问题: ${userMessage}

请基于项目上下文回答。
`;

  // 3. 调用LLM
  return await callLLM(enhancedPrompt);
}
```

### 示例2: 一键代码生成和测试

```javascript
async function generateAndTest(description, language) {
  // 1. 生成代码
  const result = await window.electronAPI.code.generate(description, {
    language: language,
    includeTests: true,
    includeComments: true,
    style: 'modern'
  });

  // 2. 自动审查
  const review = await window.electronAPI.code.review(result.code, language);

  // 3. 如果评分低，使用改进版本
  const finalCode = review.score < 8
    ? review.improvedCode
    : result.code;

  return {
    code: finalCode,
    tests: result.tests,
    score: review.score,
    suggestions: review.suggestions
  };
}
```

### 示例3: 视频自动处理流程

```javascript
async function processVideo(videoPath) {
  // 1. 获取视频信息
  const info = await window.electronAPI.video.getInfo(videoPath);
  console.log('视频时长:', info.duration);

  // 2. 剪辑视频 (前60秒)
  await window.electronAPI.video.trim({
    inputPath: videoPath,
    outputPath: '/output/trimmed.mp4',
    startTime: 0,
    duration: 60
  });

  // 3. 生成字幕
  const subtitles = await window.electronAPI.video.generateSubtitles({
    videoPath: '/output/trimmed.mp4',
    language: 'zh'
  });

  // 4. 添加字幕
  await window.electronAPI.video.addSubtitles({
    videoPath: '/output/trimmed.mp4',
    srtPath: subtitles.srtPath,
    outputPath: '/output/final.mp4'
  });

  // 5. 压缩视频
  await window.electronAPI.video.compress({
    inputPath: '/output/final.mp4',
    outputPath: '/output/compressed.mp4',
    quality: 'medium'
  });
}
```

### 示例4: 设置每日自动化工作流

```javascript
async function setupDailyWorkflow(projectId) {
  // 每天晚上9点自动生成项目报告
  await window.electronAPI.automation.createRule({
    projectId: projectId,
    name: '每日项目报告',
    description: '每天21:00自动生成项目进度报告',
    triggerType: 'schedule',
    triggerConfig: {
      cron: '0 21 * * *'
    },
    actionType: 'generate_report',
    actionConfig: {
      reportType: 'daily',
      outputPath: '/reports/daily/'
    }
  });

  // 文件变化时自动提交Git
  await window.electronAPI.automation.createRule({
    projectId: projectId,
    name: 'Git自动提交',
    description: '检测到代码变化时自动提交',
    triggerType: 'file_change',
    triggerConfig: {
      path: '/path/to/project',
      pattern: '**/*.{js,ts,vue}',
      events: ['change']
    },
    actionType: 'git_commit',
    actionConfig: {
      projectPath: '/path/to/project',
      commitMessage: 'Auto commit: {{timestamp}}',
      autoPush: false
    }
  });
}
```
