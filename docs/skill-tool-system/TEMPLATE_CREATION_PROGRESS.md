## 📝 模板创建参考指南

基于已完成的13个模板，您可以参照以下模式快速创建新模板：

### 模板结构标准

```json
{
  "id": "tpl_[category]_[name]_001",
  "name": "[english_name]",
  "display_name": "[中文显示名称]",
  "description": "[简短描述，1行]",
  "icon": "[emoji]",
  "category": "[分类名]",
  "subcategory": "[子分类]",
  "tags": ["标签1", "标签2", "标签3"],
  "project_type": "document|app|data",
  "prompt_template": "[详细的提示词模板，使用Handlebars语法]",
  "variables_schema": [
    {
      "name": "variableName",
      "label": "变量标签",
      "type": "string|number|select|textarea",
      "required": true|false,
      "default": "默认值",
      "placeholder": "占位提示",
      "options": [...]  // 仅select类型
    }
  ],
  "file_structure": {
    "root": "{{projectName}}",
    "files": [...]
  },
  "is_builtin": true,
  "author": "ChainlessChain Team",
  "version": "1.0.0",
  "usage_count": 0,
  "rating": 0,
  "rating_count": 0
}
```

### 变量类型示例

```javascript
// 字符串
{"name": "title", "type": "string", "required": true}

// 数字
{"name": "duration", "type": "number", "min": 1, "max": 60, "default": 10}

// 选择
{"name": "type", "type": "select", "options": [
  {"value": "A", "label": "选项A"},
  {"value": "B", "label": "选项B"}
]}

// 文本域
{"name": "description", "type": "textarea", "placeholder": "详细描述"}
```

### Handlebars语法参考

```handlebars
{{! 变量输出 }}
{{variableName}}

{{! 条件判断 }}
{{#if condition}}
  内容
{{else}}
  其他内容
{{/if}}

{{! 相等判断 }}
{{#eq value "expected"}}
  匹配内容
{{/eq}}

{{! 循环 }}
{{#each items}}
  {{this}}
{{/each}}

{{! 数学运算 }}
{{add number 10}}
{{subtract number 5}}
{{multiply number 2}}
{{divide number 3}}

{{! 比较 }}
{{#if (gt number 10)}}大于10{{/if}}
{{#if (lt number 10)}}小于10{{/if}}
```
